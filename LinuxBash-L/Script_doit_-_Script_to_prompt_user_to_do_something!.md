```bash
#!/bin/bash
help_text="
NAME
  doit - Prompt to do something and create a note of what got done.

USAGE
  doit [options] 

OPTIONS
  -g
    To signify this is being run from git-bash

  -h|--help
    Show help text.

  -l|--list
    Display the list of things I did.

  -s|--schedule
    Display the timed events summary (timed event is text in the 
    format XXX:HH - XXX - any characters, HH - hours.

DESCRIPTION
  I always have a list of things (small things) I need to do.

AUTHOR
  mjnurse.dev - 2023
"

help_line="Script to prompt user to do something!"
web_desc_line="Script to prompt user to do something!"

if [[ "$GITBASH" == "Y" || "$1" == "-g" ]]; then
  i_did="/c/MJN/drive/i-did.txt"
else
  i_did="/mnt/c/MJN/drive/i-did.txt"
fi

try="Try ${0##*/} -h for more information"
tmp="${help_text##*USAGE}"
usage=$(echo "Usage: ${tmp%%OPTIONS*}" | tr -d "\n" | sed "s/  */ /g")

case $1 in
  -h|--help|?)
    echo "$help_text"
    exit
    ;;
  -l|--list)
    cat ${i_did}
    exit
    ;;
  -s|--schedule)
    tmpfile=/tmp/dme
    cat $i_did | grep -o " [^ ]*:.\|^........" > $tmpfile
    dte=""
    out=""
    pdow=0
    while read line; do
      if [[ $line =~ ../../.. ]]; then
        if [[ $line != $dte ]]; then
          dow="$(date -d  ${line//\//-} +'%u')"
          echo $out
          if [[ $dow -lt $pdow ]]; then
            echo "----------"
          fi
          pdow=$dow
          out="$line - "
          dte=$line
        fi
      else
        out="$out $line"
      fi
    done < $tmpfile
    echo $out
    rm -rf $tmpfile
    exit
    ;;
esac

clear

dte="$(date +'%y/%m/%d-%H:00')"

repeat_yn=y

while [[ ${repeat_yn} == y ]]; do
  echo
  read -p "What did you do? " did
  echo
  echo "${dte} - ${did}"
  echo
  read -p "Save [yN] " yn
  if [[ $yn == y ]]; then
    repeat_yn=n
  fi
done

echo "${dte} - ${did}" >> ${i_did}

echo
cat ${i_did}

echo
read -p "Press RTN to close (e RTN to edit I did, s RTN to see schedule) " v

case $v in
  e)
    vi ${i_did}
    ;;
  s)
    $0 -s
    echo 
    read -p "Press RTN to close" w
    ;;
esac
```
