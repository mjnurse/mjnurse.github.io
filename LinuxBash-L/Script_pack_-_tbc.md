```bash
#!/bin/bash
help_text="
NAME
  pack - One line description.

USAGE
  pack [options] <parameters>

OPTIONS
  -x
     Description...

  -h|--help
    Show help text.

DESCRIPTION
  Description description description description.

AUTHOR
  mjnurse.dev - 2020
"
help_line="tbc"
desc_line="tbc"

try="Try ${0##*/} -h for more information"
tmp="${help_text##*USAGE}"
usage=$(echo "Usage: ${tmp%%OPTIONS*}" | tr -d "\n" | sed "s/  */ /g")

tmp=/tmp/pack.tmp
line="-------------------------------------------------"; line="# $line$line"

if [[ "$1" == "" ]]; then
  echo "${usage}"
  echo "${try}"
  exit 1
fi

if [[ "$1" == "--help" || "$1" == "-h" || "$1" == "?" ]]; then
   echo "$help_text"
   exit
fi

if [[ $# == 1 ]]; then
  multi_yn=n
  out="$1.pack"
  rm -f $out
else
  multi_yn=y
  multi_yn=n
  out="multi.pack"
  rm -f $out
  echo "$line" >> $out
  files="$*"
  echo "ITINERARY: ${files// /, }" | fold -w 100 -s >> $out
  echo "$line" >> $out
fi


for f in $*; do
  if [[ "$f" != "pack" ]]; then
    echo "FILE=$f; echo '" >> $out
    cat $f | gzip -f | base64 --wrap=100 >> $out
    echo "' | base64 -d | gunzip > $f; chmod u+x $f" >> $out
  fi
done

cat $out
```
