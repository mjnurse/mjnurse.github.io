```bash
#!/bin/bash
help_text="
NAME
   github-badchars - One line description.

USAGE
   github-badchars [options] <parameters>

OPTIONS
   -x
      Description...

   -h|--help
      Show help text.

DESCRIPTION
   Description description description description.

AUTHOR
  mjnurse.dev - 2020
"
help_line="tbc"
desc_line="tbc"

try="Try ${0##*/} -h for more information"
tmp="${help_text##*USAGE}"
usage=$(echo "Usage: ${tmp%%OPTIONS*}" | tr -d "\n" | sed "s/  */ /g")

if [[ "$1" == "" ]]; then
  echo "${usage}"
  echo "${try}"
  exit 1
fi

if [[ "$1" == "--help" || "$1" == "-h" || "$1" == "?" ]]; then
   echo "$help_text"
   exit
fi

for f in $(find $PWD -type f); do
  case $f in
    *.db|*.jpg|*.png|*.csv|*/node_modules/*|*badchars)
      ;;
    */.git*)
      ;;
   *) 
      echo FILE: $f
      chars="$(cat $f | tr -d "[:alnum:]\n \-\"\*\#!Ł$%^&()_+-={}[]<>,.\\\|/?'\`@~\t")"
      if [[ "$chars" != "" ]]; then
        echo -e "$f |$chars|" | sed "s//^M/g"
      fi
      ;;
  esac
done

echo END ---
```
