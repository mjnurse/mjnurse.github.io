#!/bin/bash

help_line="tbc"
web_desc_line="Example Bash Script"

help_text="
usage: <<FILENAME>> [options] <filename>
-h : This help text.
"

case $1 in
    -h|--help)
        echo "$help_text"
        exit
        ;;
esac

if [[ "$1" == "" ]]; then
    echo "$help_text"
    exit 1
fi
