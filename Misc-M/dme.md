```
#!/bin/bash

set -e

cd /

## ==============================================
## Configure dmcrypt - create an encrypted volume
## ==============================================

echo "Create a script to download and decrypt the dmcrypt key (Luks key)"
mkdir -p /etc/luks

echo '#!/bin/bash

  # get encrypted key from S3
  aws s3 cp s3://mjn-general/mjn-encrypted-dmcrypt-key . > /dev/null

  # Decrypt the password from the file with KMS, save the secret password in dmcryptKey
  aws --region eu-west-2 kms decrypt --ciphertext-blob fileb://mjn-encrypted-dmcrypt-key --output text --query Plaintext
' > /etc/luks/key.sh

echo "Make this executable by root only"
chmod 500 /etc/luks/key.sh

echo "Create an empty 15GB file called encryptedfs (Encrypted File System)."
dd of=encryptedfs bs=1G count=0 seek=15

echo "Associate a loopback device with the encryptedfs file."
losetup /dev/loop9 encryptedfs

echo "Create a directory to mount the encrypted file system"
mkdir -p /mnt/encryptedfs

echo "Create a new group to control access to the encrypted file system"
groupadd -f encfsaccess

echo "Change the group owner of the encrypted file system to the new group"
chgrp encfsaccess /mnt/encryptedfs

echo "Encrypt to file system"
/etc/luks/key.sh | cryptsetup luksFormat /dev/loop9

echo "Open the file system / partition"
/etc/luks/key.sh | cryptsetup luksOpen /dev/loop9 encryptedfs

echo "Set a file system for the encryptedfs"
mkfs.ext4 -F /dev/mapper/encryptedfs

echo "Create a systemd service to unlock the encryptedfs after a reboot"
echo "
[Unit]
Description=Open encrypted data volume
After=network-online.target
Wants=network-online.target
StopWhenUnneeded=true

[Service]
Type=oneshot
ExecStart=/bin/sh -c 'losetup /dev/loop9 encryptedfs; /etc/luks/key.sh | /sbin/cryptsetup luksOpen /dev/loop9 encryptedfs'
RemainAfterExit=true
ExecStop=/sbin/cryptsetup -d - -v luksClose encryptedfs
" > /etc/systemd/system/unlock-encryptedfs.service

echo "Create a systemd service to mount the encryptedfs after a reboot"
echo "
[Unit]
Requires=unlock-encryptedfs.service
After=unlock-encryptedfs.service

[Mount]
What=/dev/mapper/encryptedfs
Where=/mnt/encryptedfs
Type=ext4
Options=defaults,noatime,_netdev

[Install]
WantedBy=multi-user.target
" > /etc/systemd/system/mnt-encryptedfs.mount

echo "Un-mount and close enctrypted fs"
umount /mnt/encryptedfs/
cryptsetup close encryptedfs
losetup -d /dev/loop9

echo "Enable and start the mount (and therefore unlock) services"
systemctl enable mnt-encryptedfs.mount
systemctl start mnt-encryptedfs.mount

# echo "Mount the encryptedfs for the initial boot"
# mount /dev/mapper/encryptedfs /mnt/encryptedfs

echo "For some reason the chgrp doesn't always work to run command again..."
chgrp encfsaccess /mnt/encryptedfs

echo "Restrict access to root / encfsaccess group"
chmod 770 /mnt/encryptedfs
```
