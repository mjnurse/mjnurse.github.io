```bash
# Docker

# -------------------------------------------------------------------------------------------------
= CONTAINERS
# -------------------------------------------------------------------------------------------------

delete container (rm) <container_name> :: docker rm $1

list containers (ps) [<-d>] :: \
    if [[ $1 == -d ]]; then docker ps --all; else docker ps --all --format "table {{.ID}}\t{{.Names}}\t{{.Status}}"; fi \
    ## -d: show details

logs (l) [<-f>] <container-name> :: \
    if [[ $1 == -f ]]; then docker logs -f $2; else docker logs $1; fi

shell (sh) [<-s>] [<-r>] <container_name> :: \
    tmp_shell=bash; tmp_user=""; \
    while [[ "$1" != "" ]]; do \
        case $1 in \
            -s) tmp_shell=sh;; \
            -r) tmp_user="--user root";; \
            *) break;; \
        esac; \
        echo $1; shift; \ 
    done; \
    docker exec -it $tmp_user $1 $tmp_shell
#    if [[ $1 == -s ]]; then docker exec -it $2 /bin/sh; else docker exec -it $1 /bin/bash; fi \
#    ## -s: Use basic 'sh' rather than 'bash'

stats (s) :: docker stats```
